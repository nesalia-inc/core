You have multiple operations that can fail. How do you chain them together without nested if-statements or try-catch blocks?

## The Problem

Traditional error handling leads to nesting hell:

```typescript
function processOrder(orderId: string): OrderResult {
  const order = findOrder(orderId);
  if (!order) {
    return { error: "Order not found" };
  }

  const customer = findCustomer(order.customerId);
  if (!customer) {
    return { error: "Customer not found" };
  }

  const payment = processPayment(customer.id, order.amount);
  if (!payment.success) {
    return { error: payment.error };
  }

  const shipment = createShipment(order, customer.address);
  return { success: true, order, shipment };
}
```

Problems:
- Deeply nested code
- Easy to forget to check for null
- Error handling is inconsistent

## The Solution: map and flatMap

### Use map for Simple Transformations

```typescript
import { ok, err, Result } from "@deessejs/core";

const result = ok(10)
  .map(x => x * 2)   // Transform the value
  .map(x => x + 5)   // Chain another transformation
  .map(x => `Result: ${x}`);

console.log(result.value); // "Result: 25"
```

If any step fails, the error passes through:

```typescript
const result = ok(10)
  .map(x => x * 2)
  .map(x => {
    if (x > 15) return err("Too big") as any;
    return x;
  })
  .map(x => x + 1);

console.log(result.error); // "Too big"
```

### Use flatMap for Operations That Might Fail

```typescript
import { ok, err, Result, flatMap } from "@deessejs/core";

const parseNumber = (s: string): Result<number, string> => {
  const n = Number(s);
  return isNaN(n) ? err("Not a number") : ok(n);
};

const divide = (a: number, b: number): Result<number, string> =>
  b === 0 ? err("Division by zero") : ok(a / b);

// Chain operations that return Results
const result = ok("20")
  .flatMap(parseNumber)    // Ok(20)
  .flatMap(n => divide(n, 2))  // Ok(10)
  .map(n => n * 3);       // Ok(30)

console.log(result.value); // 30
```

<Callout type="success">
  Without `flatMap`, you'd get nested Results: `Result<Result<Result<number>, string>, string>`.
</Callout>

## Real-World Example

A complete order processing flow:

```typescript
import { ok, err, Result, flatMap, map } from "@deessejs/core";

interface Order { id: string; amount: number; customerId: string }
interface Customer { id: string; name: string; creditLimit: number }
interface Payment { success: boolean; transactionId?: string }
interface Shipment { trackingNumber: string }

const validateOrder = (order: Order): Result<Order, string> =>
  order.amount > 0 ? ok(order) : err("Invalid amount");

const checkCredit = (customer: Customer, amount: number): Result<Customer, string> =>
  customer.creditLimit >= amount
    ? ok(customer)
    : err("Insufficient credit");

const processPayment = (customerId: string, amount: number): Result<Payment, string> => {
  // Simulate payment processing
  return ok({ success: true, transactionId: "tx_123" });
};

const createShipment = (order: Order): Result<Shipment, string> => {
  return ok({ trackingNumber: "TRACK123" });
};

const processOrderFlow = (order: Order): Result<{ order: Order; payment: Payment; shipment: Shipment }, string> => {
  return ok(order)
    .flatMap(validateOrder)
    .flatMap(o => findCustomer(o.customerId).flatMap(c => checkCredit(c, o.amount).map(() => ({ order: o, customer: c }))))
    .flatMap(({ order, customer }) => processPayment(customer.id, order.amount).map(payment => ({ order, payment })))
    .flatMap(({ order, payment }) => createShipment(order).map(shipment => ({ order, payment, shipment })));
};

// Find order from database, then chain
const orderResult = findOrder("123")
  .flatMap(processOrderFlow);

orderResult.match(
  ({ order, payment, shipment }) => {
    console.log(`Order ${order.id} processed!`);
    console.log(`Transaction: ${payment.transactionId}`);
    console.log(`Shipment: ${shipment.trackingNumber}`);
  },
  (error) => {
    console.error("Order failed:", error);
  }
);
```

## Combining Multiple Results

Use `map2`, `map3` to combine independent operations:

```typescript
import { ok, err, map2, map3, Result } from "@deessejs/core";

const fetchUser = (id: string): Result<User, string> => { /* ... */ };
const fetchProduct = (id: string): Result<Product, string> => { /* ... */ };

const orderResult = map2(
  fetchUser("123"),
  fetchProduct("456"),
  (user, product) => ({
    user,
    product,
    total: product.price * order.quantity
  })
);
```

## When to Use map vs flatMap

| Method | Use When |
|--------|----------|
| `map` | Transformation doesn't fail |
| `flatMap` | Transformation can fail |

```typescript
// Using map - the function returns a plain value
ok(10).map(x => x * 2);  // Ok(20)

// Using flatMap - the function returns a Result
ok("10").flatMap(s => parseNumber(s));  // Ok(10) or Err(...)
```

## Recap

- Use `map` for simple value transformations
- Use `flatMap` when your transformation can fail
- Avoid nested Results with `flatMap`
- Use `map2`, `map3` to combine independent operations
